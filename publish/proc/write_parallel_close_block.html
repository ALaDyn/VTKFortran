   function write_parallel_close_block(self) result(error)
   !< Close a block container.
   class(xml_writer_abstract), intent(inout) :: self  !< Writer.
   integer(I4P)                              :: error !< Error status.

   self%vtm_block(2) = -1
   call self%write_end_tag(name='Block')
   error = self%error
