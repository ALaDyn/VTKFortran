   elemental function byte_size_R4P(i) result(bytes)
   !< Compute the number of bytes of a real variable.

   integer(I1P)          :: bytes !< Number of bytes of r.

   bytes = bit_size(i) / 8_I1P
   endfunction byte_size_R4P
