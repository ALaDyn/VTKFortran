00704   pure subroutine encode_bits(bits,padd,code)
00705   !---------------------------------------------------------------------------------------------------------------------------------
00706   implicit none
00707   integer(I1P), intent(IN)::  bits(1:)  !< Bits to be encoded.
00708   integer(I4P), intent(IN)::  padd      !< Number of padding characters ('=').
00709   character(1), intent(OUT):: code(1:)  !< Characters code.
00710   integer(I1P)::              sixb(1:4) !< 6 bits slices (stored into 8 bits integer) of 24 bits input.
00711   integer(I8P)::              c,e       !< Counters.
00712   !---------------------------------------------------------------------------------------------------------------------------------
00713 
00714   !---------------------------------------------------------------------------------------------------------------------------------
00715   c = 1_I8P
00716   do e=1_I8P,size(bits,dim=1),3_I8P ! loop over array elements: 3 bytes (24 bits) scanning
00717     sixb = 0_I1P
00718     call mvbits(bits(e  ),2,6,sixb(1),0)
00719     call mvbits(bits(e  ),0,2,sixb(2),4) ; call mvbits(bits(e+1),4,4,sixb(2),0)
00720     call mvbits(bits(e+1),0,4,sixb(3),2) ; call mvbits(bits(e+2),6,2,sixb(3),0)
00721     call mvbits(bits(e+2),0,6,sixb(4),0)
00722     sixb = sixb + 1_I1P
00723     code(c  :c  )(1:1) = base64(sixb(1):sixb(1))
00724     code(c+1:c+1)(1:1) = base64(sixb(2):sixb(2))
00725     code(c+2:c+2)(1:1) = base64(sixb(3):sixb(3))
00726     code(c+3:c+3)(1:1) = base64(sixb(4):sixb(4))
00727     c = c + 4_I8P
00728   enddo
00729   if (padd>0) code(size(code,dim=1)-padd+1:)(1:1)='='
00730   return
00731   !---------------------------------------------------------------------------------------------------------------------------------
00732   endsubroutine encode_bits
